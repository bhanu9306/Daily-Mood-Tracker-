from flask import Flask, render_template, request, redirect
import sqlite3
from datetime import datetime
import random

app = Flask(__name__)

# Quotes based on mood
quotes = {
    "Happy": [
        "Keep smiling, life is beautiful!",
        "Happiness looks good on you!",
        "Enjoy every moment!"
    ],
    "Sad": [
        "Tough times never last, but tough people do.",
        "Every day is a fresh start.",
        "You are stronger than you think."
    ],
    "Angry": [
        "Take a deep breath, calm your mind.",
        "Peace begins with you.",
        "Let go of what you can't control."
    ],
    "Tired": [
        "Rest if you must, but don’t quit.",
        "Take care of your body and mind.",
        "Small steps still move you forward."
    ]
}

# Initialize DB
def init_db():
    conn = sqlite3.connect("mood.db")
    c = conn.cursor()
    c.execute("""
        CREATE TABLE IF NOT EXISTS moods (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            mood TEXT,
            date TEXT
        )
    """)
    conn.commit()
    conn.close()

# Home page
@app.route("/")
def index():
    conn = sqlite3.connect("mood.db")
    c = conn.cursor()
    c.execute("SELECT * FROM moods ORDER BY id DESC")
    data = c.fetchall()
    conn.close()
    return render_template("index.html", data=data)

# Add mood
@app.route("/add", methods=["POST"])
def add():
    mood = request.form["mood"]
    date = datetime.now().strftime("%Y-%m-%d %H:%M")

    conn = sqlite3.connect("mood.db")
    c = conn.cursor()
    c.execute("INSERT INTO moods (mood, date) VALUES (?, ?)", (mood, date))
    conn.commit()
    conn.close()

    # Get random quote
    quote = random.choice(quotes[mood])
    return render_template("index.html", quote=quote)

if __name__ == "__main__":
    init_db()
    app.run(debug=True)
